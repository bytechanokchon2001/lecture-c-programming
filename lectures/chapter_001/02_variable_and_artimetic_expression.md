# Variable and Artimetic expression
เรามาลองเขียนโปรแกรมสำหรับแปรงอุณหภูมิจากฟาเรนไฮต์เป็นเซลเซียสกัน โดยเราสามารถใช้สูตร `C = (5/9)(F-32)`

    /*
        แสดงตารางการแปรงอุณหภูมิจาก ฟาเรนไฮต์ เป็น เซลเซียส 
        ตั้งแต่ 0F ถึง 300F
    */
    int main()
    {
        int fahr, celsius;
        int lower, upper, step;

        lower = 0;      // อุณหภูมิต่ำสุด
        upper = 300;    // อุณหภูมิสูงสุด
        step = 20;      // จำนวนอุณหภูมิที่เพิ่มต่อครั้ง

        fahr = lower;

        while (fahr <= upper)
        {
            celsius = 5 * (fahr - 32) / 9;
            printf("%d\t%d\n", fahr, celsius);
            fahr = fahr + step;
        }

        return 0;
    }

## อธิบายชุดคำสั่ง
### คอมเมนท์
    /*
        แสดงตารางการแปรงอุณหภูมิจาก ฟาเรนไฮต์ เป็น เซลเซียส 
        ตั้งแต่ 0F ถึง 300F
    */

สิ่งนี้เรียกว่า `Comment` มักใช้อธิบายการทำงานของโปรแกรม เพื่อให้โปรแกรมสามารถทำความเข้าใจได้ง่ายขึ้น โดยการคอมเมนท์จะถูกแบ่งออกเป็น 2 แบบ คือ
- `/* TEXT */` ใช้สำหรับการคอมเมนท์แบบหลายบรรทัด
- `// TEXT` ใช้สำหรับการคอมเมนท์บรรทัดเดียว

### การประกาศตัวแปร
    int fahr, celsius;
    int lower, upper, step;

ในภาษา C ตัวแปรทั้งหมดจะต้องถูกประกาศก่อนการใช้งาน โดยปกติจะประกาศไว้ที่ต้นของฟังก์ชันก่อนการทำงานใด ๆ
โดยชนิดของข้อมูลในภาษา C มีดังนี้
- `integer` ใช้จัดเก็บเลขจำนวนเต็ม ไม่มีทศนิยม *มีขนาด 4 byte* โดยสามารถใช้สัญลักษณ์ ``
- `float` ใช่จัดเก็บเลขที่มีทศนิยม *มีขนาด 4 byte*
- `double` ใช้จัดเก็บเลขที่มีทศนิยม *มีขนาด 8 byte*
- `character` ใช้จัดเก็บอักขระตัวอักษร *มีขนาด 1 byte*

### การกำหนดค่า

    lower = 0;
    upper = 300;
    step = 20;

การคำนวนของโปรแกรม เริ่มต้นด้วยการกำหนดค่า ซึ่งเป็นค่าเริ่มต้นของตัวแปร โดยทุกคำสั่งจะต้องจบด้วย `semicolon`

### การวนซ้ำ

    while (faht <= upper) 
    {
        // do something
    }

แต่ละบรรทัดของตารางจะใช้วิธีการคำนวนเดียวกัน เราจึงใช้วิธีการลูปเพื่อคำนวนแต่ละค่า
โดยการทำงานของ `while` คือ หากเงื่อนไขเป็นจริง จะทำงานชุดคำสั่งด้านใน โดยเงื่อนไขในที่นี้คือ `faht <= upper`

งานส่วนใหญ่จะเกิดขึ้นภายในลูป โดยผลลัพธ์จะถูกคำนวนที่ส่วน `celsius = 5 * (fahr - 32) / 9;`

### การแสดงผล

    printf("%d\t%d\n", fahr, celsius);

จากชุดคำสั่งดังกล่าว จะพบการใช้งานของฟังก์ชัน `printf` อีกรูปแบบ โดยการจะใส่ข้อมูลจากตัวแปรเข้าไปที่สตริง เราจะต้องระบุ `%` เพื่อกำหนดตำแหน่งที่ต้องการให้ข้อมูลแสดงผล เช่น `%d` หมายถึง ในตำแหน่งนั้นของสตริงจะถูกแทนที่ด้วยเลขจำนวนเต็ม

ตอนนี่เราพบปัญหา ตัวเลขไม่ได้แสดงผลชนิดขวา เราสามารถเพิ่มช่องว่างให้กับแต่ละตำแหน่งได้ เช่น `%3d`, `%6d`

การกำหนดข้อมูลที่จะแสดงผลในสตริงสำหรับ decimal มีดังนี้
- `%d` แสดงผลในรูปแบบเลขจำนวนเต็ม
- `%6d` แสดงผลเลขจำนวนเต็ม โดยสามารถแสดงได้มากที่สุด 6 ตัว
- `%f` แสดงผลเป็นเลขทศนิยม
- `%6f` แสดงผลเป็นเลขทศนิยม โดยสามารถแสดงได้มากที่สุด 6 ตัว
- `%.2f` แสดงผลเป็นเลขทศนิยม โดยกำหนดให้ทศนิยมถูกแสดง 2 ตำแหน่ง
- `%6.2d` แสดงผลเป็นเลขทศนิยม โดยสามารถแสดงได้มากที่สุด 6 ตัว และมีทศนิยม 2 ตำแหน่ง
